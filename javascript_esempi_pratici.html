function aggiungiTodo() {
    const input = document.getElementById('nuovoTodo');
    const testo = input.value.trim();
    
    if (testo === '') return;
    
    const lista = document.getElementById('listaTodo');
    const li = document.createElement('li');
    li.innerHTML = testo + ' <button onclick="rimuoviTodo(this)">X</button>';
    lista.appendChild(li);
    
    input.value = '';
}

function rimuoviTodo(button) {
    button.parentElement.remove();
}

    JavaScript
    let cronologia = ['home'];
let paginaCorrente = 0;

function navigaA(nuovaPagina) {
    // Aggiungi alla cronologia
    cronologia.push(nuovaPagina);
    paginaCorrente = cronologia.length - 1;
    
    aggiornaBreadcrumb();
    aggiornaContenuto();
}

function tornaIndietro() {
    if (paginaCorrente > 0) {
        paginaCorrente--;
        aggiornaBreadcrumb();
        aggiornaContenuto();
    }
}

function aggiornaBreadcrumb() {
    const breadcrumb = document.getElementById('breadcrumb');
    const percorso = cronologia.slice(0, paginaCorrente + 1);
    
    const icone = {
        'home': 'üè†',
        'prodotti': 'üì¶',
        'categoria': 'üìÇ',
        'dettaglio': 'üìÑ'
    };
    
    breadcrumb.innerHTML = percorso.map(pagina => 
        icone[pagina] + ' ' + pagina.charAt(0).toUpperCase() + pagina.slice(1)
    ).join(' > ');
}

function aggiornaContenuto() {
    const contenuti = {
        'home': 'Benvenuto nella home page',
        'prodotti': 'Lista dei nostri prodotti',
        'categoria': 'Prodotti della categoria selezionata',
        'dettaglio': 'Dettagli del prodotto specifico'
    };
    
    const paginaAttuale = cronologia[paginaCorrente];
    document.getElementById('contenutoBreadcrumb').innerHTML = contenuti[paginaAttuale];
}

    function apriModal() {
    const modal = document.getElementById('modalSlide');

    setTimeout(function() {
        modal.children[0].style.transform = 'translateY(0)';
    }, 10);

function chiudiModal() {
    const modal = document.getElementById('modalSlide');
    modal.children[0].style.transform = 'translateY(-100%)';
    setTimeout(function() {
        modal.style.display = 'none';
    }, 500);
}

// Chiudi cliccando fuori

    const modal = document.getElementById('modalSlide');
